import type { ChainModifiers, Entry, EntryFieldTypes, EntrySkeletonType, LocaleCode } from "contentful";

export interface TypeBrandFields {
    companyName: EntryFieldTypes.Text;
    logo?: EntryFieldTypes.AssetLink;
    companyDescription?: EntryFieldTypes.Text;
    website?: EntryFieldTypes.Symbol;
    twitter?: EntryFieldTypes.Symbol;
    email?: EntryFieldTypes.Symbol;
    phone?: EntryFieldTypes.Array<EntryFieldTypes.Symbol>;
}

export type TypeBrandSkeleton = EntrySkeletonType<TypeBrandFields, "brand">;
export type TypeBrand<Modifiers extends ChainModifiers, Locales extends LocaleCode = LocaleCode> = Entry<TypeBrandSkeleton, Modifiers, Locales>;

export interface TypeCategoryFields {
    title: EntryFieldTypes.Text;
    icon?: EntryFieldTypes.AssetLink;
    categoryDescription?: EntryFieldTypes.Text;
}

export type TypeCategorySkeleton = EntrySkeletonType<TypeCategoryFields, "category">;
export type TypeCategory<Modifiers extends ChainModifiers, Locales extends LocaleCode = LocaleCode> = Entry<TypeCategorySkeleton, Modifiers, Locales>;

export interface TypeProductFields {
    productName: EntryFieldTypes.Text;
    slug?: EntryFieldTypes.Symbol;
    productDescription?: EntryFieldTypes.Text;
    sizetypecolor?: EntryFieldTypes.Symbol;
    image?: EntryFieldTypes.Array<EntryFieldTypes.AssetLink>;
    tags?: EntryFieldTypes.Array<EntryFieldTypes.Symbol>;
    categories?: EntryFieldTypes.Array<EntryFieldTypes.EntryLink<TypeCategorySkeleton>>;
    price?: EntryFieldTypes.Number;
    brand?: EntryFieldTypes.EntryLink<TypeBrandSkeleton>;
    quantity?: EntryFieldTypes.Integer;
    sku?: EntryFieldTypes.Symbol;
    website?: EntryFieldTypes.Symbol;
}

export type TypeProductSkeleton = EntrySkeletonType<TypeProductFields, "product">;
export type TypeProduct<Modifiers extends ChainModifiers, Locales extends LocaleCode = LocaleCode> = Entry<TypeProductSkeleton, Modifiers, Locales>;
